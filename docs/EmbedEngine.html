    'use strict';

let fs = require('fs');
let URLEmbedProvider = require('./URLEmbedProvider');
let OEmbedProvider = require('./OEmbedProvider');

/**
 * this is EmbedEngine description.
 */
class EmbedEngine {

  /**
   * this is EmbedEngine constructor description.
   * @param {string} [name="anonymous"] - this is name description.
   */
  constructor (options) {
    this.providerRegistry = [];
    this.defaultProviderClasses = [];
    this.options = options;
  }

  registerProvider (provider) {
    this.configureProvider(provider);
    this.providerRegistry.push(provider);
  }

  configureProvider (provider) {
    provider.configure(this.options);
  }

  registerDefaultProviders (path) {
    this.defaultProviderClasses = this.loadDefaultProviders(path);

    for (let i = 0; i < this.defaultProviderClasses.length; i++) {
      let ProviderClass = this.defaultProviderClasses[i];
      let provider = new ProviderClass();
      this.configureProvider(provider);
      this.registerProvider(provider);
    }
  }

  loadDefaultProviders (path) {
    if (!path) path = this.defaultProviderDirectory;
    let providerClasses = [];

    let files = fs.readdirSync(path);
    for (let i = 0; i < files.length; i++) {
      let providerClass = require(path + '/' + files[i]);
      providerClasses.push(providerClass);
    }

    return providerClasses;
  }

  getEmbed (options, callback) {
    if (!options || !options.embedURL) {
      throw new Error('Options object missing required property: embedURL');
    }

    let embedURL = options.embedURL;
    let providerFound = false;

    for (let idx = 0; idx < this.providerRegistry.length; idx++) {
      let provider = this.providerRegistry[idx];
      let self = this;

      if (provider.isMatch(embedURL)) {
        providerFound = true;

        let d = new Date();

        provider.getEmbed(options, function (error, data) {
          data.elapsed = (new Date()).getTime() - d.getTime();
          if (error) {
            callback(error, data);
          } else {
            self.filterMarkup(data);
            callback(null, data);
          }
        });

        idx = this.providerRegistry.length;
      }
    }

    if (!providerFound) {
      callback(new Error('Unknown embed provider for url: ' + embedURL));
    }
  }

  filterMarkup (data) {
    return data;
  }
}

EmbedEngine.prototype.defaultProviderDirectory = __dirname + '/../default_providers';

EmbedEngine.prototype.providerSuperClasses = {
  URLEmbedProvider: URLEmbedProvider,
  OEmbedProvider: OEmbedProvider
};

module.exports = EmbedEngine;
