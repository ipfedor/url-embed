    'use strict';

class URLEmbedProvider {
  constructor (urlPatterns) {
    if (urlPatterns) this.urlPatterns = urlPatterns;
  }

  isMatch (embedURL) {
    let matches = false;
    for (let idx = 0; idx < this.urlPatterns.length; idx++) {
      if (embedURL.match(this.urlPatterns[idx])) {
        matches = true;
        idx = this.urlPatterns.length;
      }
    }
    return matches;
  }

  getEmbed (options, callback) {
    if (!options || !options.embedURL) {
      throw new Error('Options object missing required property: embedURL');
    }

    let data = {
      html: this.STUB_MARKUP,
      options: options
    };

    try {
      this.filterData(data);
      callback(null, data);
    } catch (error) {
      data = {
        html: this.errorMarkup(options, error)
      };
      callback(error, data);
    }
  }

  filterData (data) {
    return data.html;
  }

  errorMarkup (options, error, errorMessage) {
    let embedURL = options.embedURL;

    if (!errorMessage) {
      errorMessage = options.embedURL;
    }
    return '<a href="' + embedURL + '">' + embedURL + '</a>';
  }

  // Stub method caled by EmbedEngine
  configure (configOptions) {

  }
}

URLEmbedProvider.prototype.STUB_MARKUP = '<!-- stub -->';

module.exports = URLEmbedProvider;
