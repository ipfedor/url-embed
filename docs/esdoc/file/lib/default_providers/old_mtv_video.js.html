    lib/default_providers/old_mtv_video.js

    'use strict';

/** @ignore */
let URLEmbedProvider = require('../classes/URLEmbedProvider');

class MTVVideoOld extends URLEmbedProvider {
  getEmbed (options, callback) {
    let embedURL = options.embedURL;

    let data = {};
    try {
      let matches = embedURL.match('/videos/misc/([0-9]+)');
      data.html = '<iframe src="http://media.mtvnservices.com/embed/mgid:uma:video:mtv.com:' + matches[1] + '" width="512" height="288" frameborder="0"></iframe>';
      this.filterData(data);
      callback(null, data);
    } catch (error) {
      data = {
        html: this.errorMarkup(embedURL)
      };
      callback(error, data);
    }
  }
}

MTVVideoOld.prototype.name = 'old_mtv_video';
MTVVideoOld.prototype.urlPatterns = ['^http://www\.mtv\.com/videos/misc/.*'];

module.exports = MTVVideoOld;
